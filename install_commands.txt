# Сначала создайте все необходимые файлы:

cat > nginx.conf << 'EOL'
server {
    listen 80;
    server_name 185.104.113.145;
    root /var/www/yurist_mini;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    location ~* \.(js|css|png|jpg|jpeg|gif|ico)$ {
        expires max;
        log_not_found off;
    }
}
EOL

cat > yurist-bot.service << 'EOL'
[Unit]
Description=Yurist Telegram Bot
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/var/www/yurist_mini
ExecStart=/usr/bin/python3 bot.py
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target
EOL

# Обновление системы
apt update && apt upgrade -y

# Установка необходимых пакетов
apt install -y nginx python3 python3-pip git

# Создание директории для сайта
mkdir -p /var/www/yurist_mini

# Клонирование репозитория
cd /var/www
rm -rf yurist_mini
git clone https://github.com/Kdebugada/yurist_mini.git

# Настройка nginx
cp nginx.conf /etc/nginx/sites-available/yurist
ln -sf /etc/nginx/sites-available/yurist /etc/nginx/sites-enabled/
rm -f /etc/nginx/sites-enabled/default
nginx -t && systemctl restart nginx

# Установка зависимостей Python
cd /var/www/yurist_mini
pip3 install -r requirements.txt

# Настройка systemd сервиса для бота
cp yurist-bot.service /etc/systemd/system/
systemctl daemon-reload
systemctl enable yurist-bot
systemctl start yurist-bot

# Настройка автоматического обновления из git
echo "*/5 * * * * cd /var/www/yurist_mini && git pull origin main" | crontab -

echo "Установка завершена!"
